TABLE: donors
CREATE TABLE donors ( donor_id VARCHAR(10) PRIMARY KEY, name VARCHAR(100) NOT NULL, age INT CHECK(age BETWEEN 18 AND 65), blood_group VARCHAR(5) NOT NULL, phone VARCHAR(15), city VARCHAR(50), last_donated DATE );
TABLE: blood_inventory
CREATE TABLE blood_inventory ( inv_id VARCHAR(10) PRIMARY KEY, blood_group VARCHAR(5) NOT NULL, units INT CHECK(units >= 0), expiry_date DATE, donor_id VARCHAR(10), FOREIGN KEY(donor_id) REFERENCES donors(donor_id) );
TABLE: blood_requests
CREATE TABLE blood_requests ( req_id VARCHAR(10) PRIMARY KEY, hospital_id VARCHAR(10), patient VARCHAR(100), blood_group VARCHAR(5), units INT, priority VARCHAR(10), status VARCHAR(10) DEFAULT 'PENDING', contact VARCHAR(15), FOREIGN KEY(hospital_id) REFERENCES hospitals(hospital_id) );
TABLE: hospitals
CREATE TABLE hospitals ( hospital_id VARCHAR(10) PRIMARY KEY, name VARCHAR(100), city VARCHAR(50), contact VARCHAR(15) );
